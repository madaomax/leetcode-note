# [54. Spiral Matrix](https://leetcode.com/problems/spiral-matrix/)

Approach 1

```python
class Solution:
    def spiralOrder(self, matrix: List[List[int]]) -> List[int]:
        """
        Summary:
            Spiral order:
            go from [0,0], traverse first row, 


        Args:
            matrix: int matrix, m * n


        Returns:
            all elem of the `matrix` in spiral order.
        
        e.g.
            Input: matrix = [[1,2,3],[4,5,6],[7,8,9]]
            Output: [1,2,3,6,9,8,7,4,5]

        
        """
        # idea: mark traversed cell as invalid, e.g. -inf
        # float("inf") as int
        i = 0
        j = 0
        m = len(matrix) # num rows
        n = len(matrix[0])
        result = []
        # spiral order: right, down, left, up. delta(i, j)
        direction = [(0, 1), (1, 0), (0, -1), (-1, 0)]
        # two ptr for index. i is row num, j is col num.
        order = 0
        while i < m and j < n and len(result) < m * n:
            result.append(matrix[i][j])
            matrix[i][j] = None # mark cell as visited
            # if we are at a "corner", need order += 1
            next_i = i + direction[order % 4][0]
            next_j = j + direction[order % 4][1]
            if next_i >= m or next_j >= n or matrix[next_i][next_j] is None:
                order += 1
            # update i, j
            i, j = i + direction[order % 4][0], j + direction[order % 4][1]
        return result
```

Space: O(mn)

time: O(mn)



## Approach 2: (preferred) Shrink boundary

```python
class Solution:
    def spiralOrder(self, matrix: List[List[int]]) -> List[int]:
        """
        Summary:
            Spiral order:
            go from [0,0], traverse first row, 


        Args:
            matrix: int matrix, m * n


        Returns:
            all elem of the `matrix` in spiral order.
        
        e.g.
            Input: matrix = [[1,2,3],[4,5,6],[7,8,9]]
            Output: [1,2,3,6,9,8,7,4,5]

        
        """
        # idea: mark traversed cell as invalid, e.g. -inf
        # float("inf") as int
        i = 0
        j = 0
        m = len(matrix) # num rows
        n = len(matrix[0])
        result = []
        # spiral order: right, down, left, up. delta(i, j)
        direction = [(0, 1), (1, 0), (0, -1), (-1, 0)]
        # two ptr for index. i is row num, j is col num.
        order = 0
        low_i = 0
        high_i = m
        low_j = 0
        high_j = n
        while low_i <= i and i < high_i and low_j <= j and j < high_j and len(result) < m * n:
            result.append(matrix[i][j])
            # if we are at a "corner", need order += 1
            next_i = i + direction[order][0]
            next_j = j + direction[order][1]
            if next_i >= high_i or next_i < low_i or next_j >= high_j or next_j < low_j:
                order = (order + 1) % 4
                # maybe can put this into an arr to make it cleaner
                if order == 1:
                    # right to down, shrink row
                    low_i += 1
                elif order == 2:
                    # down to left, shrink right col
                    high_j -= 1
                elif order == 3:
                    # left to up, shrink bottom row
                    high_i -= 1
                else:
                    # up to right, shrink left col  
                    low_j += 1
            # update i, j
            i, j = i + direction[order][0], j + direction[order][1]
        return result
```

time: O(mn)

space: O(1)

```python
class Solution:
    def spiralOrder(self, matrix: List[List[int]]) -> List[int]:
        """
        Summary:
            Spiral order:
            go from [0,0], traverse first row, 


        Args:
            matrix: int matrix, m * n


        Returns:
            all elem of the `matrix` in spiral order.
        
        e.g.
            Input: matrix = [[1,2,3],[4,5,6],[7,8,9]]
            Output: [1,2,3,6,9,8,7,4,5]

        
        """
        result = []
        low_i = 0
        high_i = len(matrix)
        low_j = 0
        high_j = len(matrix[0])
        while low_i < high_i and low_j < high_j:
            for j in range(low_j, high_j):
                result.append(matrix[low_i][j])
            low_i += 1 # right to down, shrink row
            for i in range(low_i, high_i):
                result.append(matrix[i][high_j-1])
            high_j -= 1  # down to left, shrink right col
            if low_i < high_i:
                for j in range(high_j-1, low_j-1, -1):
                    result.append(matrix[high_i-1][j])
                high_i -= 1 # left to up, shrink bottom row
            if low_j < high_j:
                for i in range(high_i-1, low_i-1, -1):
                    result.append(matrix[i][low_j])
                low_j += 1 # up to right, shrink left col
        return result
```
