# [981. Time Based Key-Value Store](https://leetcode.com/problems/time-based-key-value-store/)

### Approach 1: One dictionary + Binary search

```python
class TimeMap:

    def __init__(self):
        self.dict = {}
        

    def set(self, key: str, value: str, timestamp: int) -> None:
        """
        Stores the key `key` with the value `value` at the given time `timestamp`.
        """
        if not self.dict.get(key, 0):
            self.dict[key] = [(value, timestamp)]
        self.dict[key].append((value, timestamp))

    def get(self, key: str, timestamp: int) -> str:
        """
        Speed up using binary search. 
        """
        if key not in self.dict:
            return ""
        res = ""
        #self.dict[key].sort(key=lambda x:x[1])
        values = self.dict[key]
        l = 0
        r = len(values) - 1
        while l <= r:
            mid = (l + r) // 2
            # values[mid] will be a pair of values and timestamps
            if values[mid][1] <= timestamp:
                # Ans found. Store value into result.
                res = values[mid][0]
                # check for closer values
                l = mid + 1
            else:
                r = mid - 1
        return res
```
